    <script>
        function toggleReparto(id) {
            const sup = document.getElementById('content-superiori');
            const uni = document.getElementById('content-universita');
            if (id === 'superiori') {
                sup.classList.toggle('hidden');
                uni.classList.add('hidden');
                if(!sup.classList.contains('hidden')) sup.scrollIntoView({ behavior: 'smooth', block: 'center' });
            } else {
                uni.classList.toggle('hidden');
                sup.classList.add('hidden');
                if(!uni.classList.contains('hidden')) uni.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }
        }

        function selectSubject(name) {
            const sel = document.getElementById('materiaSelect');
            if(name.includes('Analisi') || name.includes('Fisica 1') || name.includes('Algebra')) {
                sel.value = 'Universita';
            } else {
                sel.value = 'Superiori';
            }
            document.getElementById('contact-area').scrollIntoView({ behavior: 'smooth' });
        }
